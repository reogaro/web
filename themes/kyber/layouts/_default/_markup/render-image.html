{{- /* Render Hook for Images: Intercepts specific formats and routes them to partials */ -}} {{- if strings.HasSuffix .Destination ".glb" -}} {{- partial "model-viewer.html" (dict "src" .Destination "alt" .Text "poster" .Title "Page" .Page) -}} {{- else if findRE `^(https?://)?(www\.)?(youtube\.com|youtu\.be)/` .Destination -}} {{- $id := replaceRE `^.*(?:youtu\.be/|v=)([^&?]+).*` `$1` .Destination -}} {{- partial "youtube-privacy.html" (dict "id" $id "Page" .Page) -}} {{- else -}} {{- $img := .Page.Resources.GetMatch .Destination -}} {{- if $img -}} {{- if and (eq $img.ResourceType "image") (ne $img.MediaType.SubType "svg") -}} {{- $ext := path.Ext .Destination -}} {{- $baseName := strings.TrimSuffix $ext .Destination -}} {{- $targetName := printf "%s-web.webp" $baseName -}} {{- $res := site.Params.imaging.inlineres | default 1600 -}} {{- $qual := site.Params.imaging.inlinequality | default 85 -}} {{- $processed := "" -}} {{- if gt $img.Width $res -}} {{- $processed = $img.Process (printf "resize %dx webp q%d" $res $qual) | resources.Copy $targetName -}} {{- else -}} {{- $processed = $img.Process (printf "webp q%d" $qual) | resources.Copy $targetName -}} {{- end -}} {{- else -}} {{- end -}} {{- else -}} {{- end -}} {{- end -}}
